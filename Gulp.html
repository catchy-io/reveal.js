gulp.task('styles', function() {
  return sass('src/styles/main.scss', { style: 'expanded' })
    .pipe(gulp.dest('dist/assets/css'))
    .pipe(rename({suffix: '.min'}))
    .pipe(gulp.dest('dist/assets/css'))
});

gulp.src([
        'vendor/components/jquery/jquery.min.js',
        'vendor/twbs/bootstrap-sass/assets/javascripts/bootstrap.min.js',
        'js/app.js'
    ])
    .pipe(concat('app.min.js))
    .pipe(gulp.dest('web/js/'));
var filename = 'med-' + Math.floor(Date.now() / 1000) + '.min.js';
fs.readFile('app/Resources/views/base.html.twig', 'utf8', function (err, data) {
    if (err) {
        return console.log(err);
    }
    var result = data.replace(/.js.med-.+\.min.js/g, '/js/' + filename);

    fs.writeFile('app/Resources/views/base.html.twig', result, 'utf8', function (err) {
        if (err) return console.log(err);
    });
});
                    

gulp.task('watch', function () {
    gulp.watch('./sass/**/*.scss', ['sass']);
});


// Default Task
gulp.task('default', ['sass', 'watch']);
